CREATE TABLE `reporting_sessions` (
  `id` int NOT NULL AUTO_INCREMENT,
  `source_session_id` int NOT NULL,
  `module_type` enum('PITLINE','SICKLINE','COMMISSIONARY','WSP','CAI') NOT NULL,
  `coach_id` int DEFAULT NULL,
  `train_id` int DEFAULT NULL,
  `asset_id` varchar(100) DEFAULT NULL,
  `user_id` int NOT NULL,
  `inspection_datetime` datetime NOT NULL,
  `status` varchar(50) NOT NULL,
  `total_master_questions` int DEFAULT '0',
  `total_deficiencies` int DEFAULT '0',
  `total_resolved` int DEFAULT '0',
  `compliance_score` decimal(5,2) DEFAULT '0.00',
  `projected_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `attempted_count` int DEFAULT '0',
  `progress_percentage` decimal(5,2) DEFAULT '0.00',
  `is_finalized` tinyint(1) DEFAULT '0',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_source_module` (`source_session_id`,`module_type`),
  KEY `idx_reporting_datetime` (`inspection_datetime`),
  KEY `idx_reporting_module` (`module_type`),
  KEY `idx_reporting_user` (`user_id`),
  KEY `idx_reporting_status` (`status`),
  KEY `idx_reporting_source_session` (`source_session_id`)
) ENGINE=InnoDB AUTO_INCREMENT=296 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
